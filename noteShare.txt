document.querySelectorAll('#spSiteHeader, #spCommandBar, #spLeftNav, #sp-appBar')
  .forEach(el => el.style.setProperty('display', 'block', 'important'));

[...document.head.querySelectorAll('style')].forEach(s => {
  if (s.innerHTML.includes('#spCommandBar') || s.innerHTML.includes('#spSiteHeader')) {
    s.remove();
  }
});
