(async () => {
  const listId = "c106a81b-5013-43d6-9639-5c0f4e51f677"; // replace with your list GUID
  const sp = window.sp; // often exposed by SPFx/PnPjs

  const items = await sp.web.lists.getById(listId).items
    .top(10)
    .select(
      "Id",
      "FileRef",
      "FileLeafRef",
      "LinkFilename",
      "Title",
      "ContentTypeId",
      "ContentType/Name",
      "OData__dlc_DocIdUrl",
      "mvSubject"
    )
    .expand("ContentType")();

  console.log(items);
})();
